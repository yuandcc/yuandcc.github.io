 def __init__(self):
    self.pe[:, 0::2] = torch.sin(position * div_term)
    self.pe[:, 1::2] = torch.cos(position * div_term)

def pe_list(self, beg : int, end : int):
    repeats = math.ceil((end - beg) / self.max_len)
    pe = self.pe.repeat(1, repeats + 1, 1)
    x1 = beg % self.max_len
    x2 = x1 + (end - beg)
    return pe[:, x1:x2]
